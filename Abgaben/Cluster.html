distortions = []

# Range of k's
K = range(2,12,1)


for k in K:
    kmeanModel = KMeans(n_clusters=k)
    kmeanModel.fit(arr)
    distortions.append(kmeanModel.inertia_)

plt.title('Optimal K', fontsize = 25)
